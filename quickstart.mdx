<Info>
  RaTeX requires Rust 1.70+ for the server-side renderer. Native platform packages (npm, Swift, Gradle, pub.dev) have no Rust prerequisite for consumers.
</Info>

<Tabs>
  <Tab title="Web (WASM)">
    The fastest way to get started. Install the npm package and drop a web component into any HTML page.

    <Steps>
      <Step title="Install the package">
        ```bash
        npm install ratex-wasm
        ```
      </Step>
      <Step title="Add fonts and the web component script">
        ```html
        <link rel="stylesheet" href="node_modules/ratex-wasm/fonts.css" />
        <script type="module" src="node_modules/ratex-wasm/dist/ratex-formula.js"></script>
        ```
      </Step>
      <Step title="Render a formula">
        Use the `<ratex-formula>` custom element anywhere in your HTML:

        ```html
        <ratex-formula latex="\frac{-b \pm \sqrt{b^2-4ac}}{2a}" font-size="48"></ratex-formula>
        <ratex-formula latex="\ce{CO2 + H2O <=> H2CO3}" font-size="32"></ratex-formula>
        ```
      </Step>
    </Steps>

    <Tip>
      See `platforms/web/README.md` in the repository for Canvas 2D usage and bundler configuration.
    </Tip>
  </Tab>

  <Tab title="iOS">
    Integrate RaTeX into a SwiftUI app using Swift Package Manager.

    <Steps>
      <Step title="Add the package dependency">
        In Xcode, go to **File → Add Package Dependencies** and enter:

        ```
        https://github.com/erweixin/RaTeX
        ```
      </Step>
      <Step title="Import and render">
        ```swift
        import RaTeX

        struct ContentView: View {
            var body: some View {
                RaTeXFormula(
                    latex: #"\frac{-b \pm \sqrt{b^2-4ac}}{2a}"#,
                    fontSize: 24
                )
            }
        }
        ```
      </Step>
    </Steps>
  </Tab>

  <Tab title="Android">
    Add RaTeX to an Android project via Gradle.

    <Steps>
      <Step title="Add the dependency">
        In your module's `build.gradle.kts`:

        ```kotlin
        implementation("io.github.erweixin:ratex-android:0.0.15")
        ```
      </Step>
      <Step title="Add the view to your layout">
        ```xml
        <io.github.erweixin.ratex.MathView
            android:id="@+id/mathView"
            android:layout_width="wrap_content"
            android:layout_height="wrap_content" />
        ```
      </Step>
      <Step title="Set the formula from Kotlin">
        ```kotlin
        binding.mathView.latex = """\frac{-b \pm \sqrt{b^2-4ac}}{2a}"""
        binding.mathView.fontSize = 24f
        ```
      </Step>
    </Steps>
  </Tab>

  <Tab title="Flutter">
    Add RaTeX to a Flutter project via pub.dev.

    <Steps>
      <Step title="Add the dependency">
        In `pubspec.yaml`:

        ```yaml
        dependencies:
          ratex_flutter: ^0.0.15
        ```

        Then run:

        ```bash
        flutter pub get
        ```
      </Step>
      <Step title="Use the widget">
        ```dart
        import 'package:ratex_flutter/ratex_flutter.dart';

        class MyWidget extends StatelessWidget {
          @override
          Widget build(BuildContext context) {
            return RaTeXWidget(
              latex: r'\frac{-b \pm \sqrt{b^2-4ac}}{2a}',
              fontSize: 28,
            );
          }
        }
        ```
      </Step>
    </Steps>
  </Tab>

  <Tab title="React Native">
    Add RaTeX to a React Native project via npm.

    <Steps>
      <Step title="Install the package">
        ```bash
        npm install ratex-react-native
        ```
      </Step>
      <Step title="Link native modules">
        For React Native 0.60+ autolinking handles this automatically. For older versions, run:

        ```bash
        npx react-native link ratex-react-native
        ```
      </Step>
      <Step title="Render a formula">
        ```tsx
        import { RaTeXView } from 'ratex-react-native';

        export default function App() {
          return (
            <RaTeXView
              latex="\frac{-b \pm \sqrt{b^2 - 4ac}}{2a}"
              fontSize={24}
            />
          );
        }
        ```
      </Step>
    </Steps>
  </Tab>

  <Tab title="Server (PNG)">
    Render formulas to PNG on the server or in CI using the `ratex-render` binary.

    <Steps>
      <Step title="Clone and build">
        ```bash
        git clone https://github.com/erweixin/RaTeX.git
        cd RaTeX
        cargo build --release
        ```
      </Step>
      <Step title="Render a formula to PNG">
        Pipe a LaTeX string to `ratex-render` via stdin:

        ```bash
        echo '\frac{1}{2} + \sqrt{x}' | cargo run --release -p ratex-render
        ```

        Chemistry notation works the same way:

        ```bash
        echo '\ce{H2SO4 + 2NaOH -> Na2SO4 + 2H2O}' | cargo run --release -p ratex-render
        ```
      </Step>
      <Step title="Render to SVG (optional)">
        To produce a standalone SVG with embedded glyph outlines (no external fonts required):

        ```bash
        echo '\int_0^\infty e^{-x^2} dx = \frac{\sqrt{\pi}}{2}' | \
          cargo run --release -p ratex-svg --features cli -- \
          --font-dir /path/to/katex/fonts --output-dir ./out
        ```
      </Step>
      <Step title="Run the test suite">
        ```bash
        cargo test --all
        ```
      </Step>
    </Steps>

    <Note>
      The `standalone` SVG feature reads KaTeX TTF files and embeds glyph outlines directly into the SVG output, producing a fully self-contained file that renders correctly without any CSS or web fonts.
    </Note>
  </Tab>
</Tabs>
